 35 (function($) {

 37     var pluginName = 'myPlugin';
 38 
 39     /* public methods ------------------------------------------------------- */
 40     var methods = {
 41 
 42         /**
 43          * @lends $.fn.myPlugin
 44          */
 45         init: function(options) {
 46 
 47             var elem = this;
 48 
 49             return this.each(function() {
 50 
 51                 /**
 52                  * settings and defaults
 53                  * using $.extend, settings modification will affect elem.data() and vive versa
 54                  */
 55                 var settings = elem.data(pluginName);
 56                 if(typeof(settings) == 'undefined') {
 57                     var defaults = elem.myPlugin('getDefaults');
 58                     settings = $.extend({}, defaults, options);
 59                 } else {
 60                     settings = $.extend({}, settings, options);
 61                 }
 62                 elem.data(pluginName, settings);
 63 
 64                 var container_id = elem.attr("id");
 65 
 66                 // simple validation
 67                 validate_input(container_id);
 68 
 69                 // bind events
 70                 elem.unbind("onCustomEvent1").bind("onCustomEvent1", settings.onCustomEvent1);
 71 
 72 
 73             });
 74 
 75         },
 76 
 77         /**
 78          * Get default values
 79          * @example $(element).myPlugin('getDefaults');
 80          * @return {Object}
 81          */
 82         getDefaults: function() {
 83             return {
 84                 opt1: 1,
 85                 opt2: 'opt2_value',
 86 
 87                 onCustomEvent1: function() {
 88                 }
 89             };
 90         },
 91 
 92         /**
 93          * Get any option set to plugin using its name (as string)
 94          * @example $(element).myPlugin('getOption', some_option);
 95          * @param opt
 96          * @return {*}
 97          */
 98         getOption: function(opt) {
 99             var elem = this;
100             return elem.data(pluginName)[opt];
101         },
102 
103         /**
104          * Get all options
105          * @example $(element).myPlugin('getAllOptions');
106          * @return {*}
107          */
108         getAllOptions: function() {
109             var elem = this;
110             return elem.data(pluginName);
111         },
112 
113         /**
114          * Set option
115          * @example $(element).myPlugin('setOption', 'option_name',  'option_value',  reinit);
116          * @param opt {string} option names
117          * @param val
118          * @param reinit {boolean}
119          */
120         setOption: function(opt, val, reinit) {
121             var elem = this;
122             elem.data(pluginName)[opt] = val;
123             if(reinit) {
124                 elem.myPlugin('init');
125             }
126         },
127 
128         /**
129          * Refresh plugin
130          * @example $(element).myPlugin('refresh');
131          * @return {*|jQuery}
132          */
133         refresh: function() {
134             var elem = this;
135             elem.myPlugin();
136         },
137 
138         /**
139          * Destroy plugin
140          * @example $(element).myPlugin('destroy');
141          * @return {*|jQuery}
142          */
143         destroy: function() {
144             return $(this).each(function() {
145                 var $this = $(this);
146 
147                 $this.removeData(pluginName);
148             });
149         }

167     /**
168      * myPlugin - short description.
169      *
170      * @class myPlugin
171      * @memberOf $.fn
172      */
173     $.fn.myPlugin = function(method) {
174 
175         // OPTIONAL
176         if(this.size() != 1) {
177             var err_msg = 'You must use this plugin (' + pluginName + ') with a unique element (at once)';
178             this.html('<span style="color: red;">' + 'ERROR: ' + err_msg + '</span>');
179             $.error(err_msg);
180         }
181 
182         // Method calling logic
183         if(methods[method]) {
184             return methods[ method ].apply(this, Array.prototype.slice.call(arguments, 1));
185         } else if(typeof method === 'object' || !method) {
186             return methods.init.apply(this, arguments);
187         } else {
188             $.error('Method ' + method + ' does not exist on jQuery.' + pluginName);
189         }
190 
191     };
